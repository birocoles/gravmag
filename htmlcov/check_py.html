4def rectangular_prisms(prisms): 

5 """ 

6 Check if rectangular prisms are well defined 

7 

8 Parameters 

9 ---------- 

10 prisms : 2d-array 

11 Array containing the boundaries of the prisms in the following order: 

12 ``s``, ``n``, ``w``, ``e``, ``top``, ``bottom``. 

13 The array must have the following shape: (``n_prisms``, 6), where 

14 ``n_prisms`` is the total number of prisms. 

15 This array of prisms must have valid boundaries. 

16 """ 

17 prisms = np.asarray(prisms) 

18 if prisms.ndim != 2: 

19 raise ValueError( 

20 "prisms ndim ({}) ".format(prisms.ndim) 

21 + "not equal to 2" 

22 ) 

23 if prisms.shape[1] != 6: 

24 raise ValueError( 

25 "Number of columns in prisms ({}) ".format(prisms.shape[1]) 

26 + "not equal to 6" 

27 ) 

28 south, north, west, east, top, bottom = tuple(prisms[:, i] for i in range(6)) 

29 err_msg = "Invalid rectangular prism(s). " 

30 bad_sn = south > north 

31 bad_we = west > east 

32 bad_bt = top > bottom 

33 if bad_sn.any(): 

34 err_msg += "The south boundary can't be greater than the north one.\n" 

35 for prism in prisms[bad_sn]: 

36 err_msg += "\tInvalid prism: {}\n".format(prism) 

37 raise ValueError(err_msg) 

38 if bad_we.any(): 

39 err_msg += "The west boundary can't be greater than the east one.\n" 

40 for prism in prisms[bad_we]: 

41 err_msg += "\tInvalid prism: {}\n".format(prism) 

42 raise ValueError(err_msg) 

43 if bad_bt.any(): 

44 err_msg += "The top boundary can't be greater than the bottom one.\n" 

45 for prism in prisms[bad_bt]: 

46 err_msg += "\tInvalid prism: {}\n".format(prism) 

47 raise ValueError(err_msg)
